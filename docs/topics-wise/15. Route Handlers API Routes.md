# Route Handlers (API Routes) - Teaching Notes

## Topic 15: Building API Endpoints in the App Router

---

# 1. Why This Topic Matters

Route Handlers let you create **API endpoints** inside your Next.js app using `route.ts` files. They replace the old Pages Router `pages/api/` pattern.

Use Route Handlers when you need REST endpoints for mobile apps, webhooks, third-party integrations, or client-side fetch from `'use client'` components. They run on the server and can access databases, secrets, and external APIs securely.

---

# 2. Creating a Route Handler

Place a `route.ts` (or `route.js`) file inside any folder in the `app` directory. The folder path becomes the URL.

```text
app/
  api/
    users/
      route.ts        → GET/POST /api/users
    users/
      [id]/
        route.ts      → GET/PUT/DELETE /api/users/:id
    health/
      route.ts        → GET /api/health
```

```tsx
// app/api/health/route.ts
import { NextResponse } from 'next/server'

export async function GET() {
  return NextResponse.json({
    status: 'ok',
    timestamp: new Date().toISOString(),
  })
}
```

| File name | Required export names |
| --------- | --------------------- |
| `route.ts` | HTTP method functions: `GET`, `POST`, `PUT`, `PATCH`, `DELETE`, `HEAD`, `OPTIONS` |

**Important:** A folder cannot contain both `page.tsx` and `route.ts` at the same level — they conflict.

# 3. HTTP Methods

Export named functions matching HTTP verbs.

```tsx
// app/api/products/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { db } from '@/lib/db'

export async function GET() {
  const products = await db.product.findMany()
  return NextResponse.json(products)
}

export async function POST(request: NextRequest) {
  const body = await request.json()

  const product = await db.product.create({
    data: {
      name: body.name,
      price: body.price,
      category: body.category,
    },
  })

  return NextResponse.json(product, { status: 201 })
}
```

```tsx
// app/api/products/[id]/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { db } from '@/lib/db'

interface RouteContext {
  params: Promise<{ id: string }>
}

export async function GET(_request: NextRequest, context: RouteContext) {
  const { id } = await context.params
  const product = await db.product.findUnique({ where: { id } })

  if (!product) {
    return NextResponse.json({ error: 'Product not found' }, { status: 404 })
  }

  return NextResponse.json(product)
}

export async function PUT(request: NextRequest, context: RouteContext) {
  const { id } = await context.params
  const body = await request.json()

  const updated = await db.product.update({
    where: { id },
    data: { name: body.name, price: body.price },
  })

  return NextResponse.json(updated)
}

export async function DELETE(_request: NextRequest, context: RouteContext) {
  const { id } = await context.params
  await db.product.delete({ where: { id } })
  return new NextResponse(null, { status: 204 })
}
```

| Method | Typical use | Success status |
| ------ | ----------- | -------------- |
| GET | Read data | 200 |
| POST | Create resource | 201 |
| PUT | Full replace | 200 |
| PATCH | Partial update | 200 |
| DELETE | Remove resource | 204 |

# 4. Reading Request Data

### JSON body

```tsx
export async function POST(request: NextRequest) {
  const body = await request.json()
  const { email, password } = body

  if (!email || !password) {
    return NextResponse.json(
      { error: 'Email and password are required' },
      { status: 400 }
    )
  }

  // process login...
  return NextResponse.json({ success: true })
}
```

### Query parameters

```tsx
// GET /api/products?category=electronics&sort=price
export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url)
  const category = searchParams.get('category')
  const sort = searchParams.get('sort') ?? 'name'

  const products = await db.product.findMany({
    where: category ? { category } : undefined,
    orderBy: { [sort]: 'asc' },
  })

  return NextResponse.json(products)
}
```

### Headers and cookies

```tsx
import { cookies, headers } from 'next/headers'

export async function GET() {
  const headersList = await headers()
  const authHeader = headersList.get('authorization')

  const cookieStore = await cookies()
  const sessionId = cookieStore.get('session')?.value

  if (!sessionId) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  return NextResponse.json({ sessionId, authHeader })
}
```

| Data source | How to read |
| ----------- | ----------- |
| JSON body | `await request.json()` |
| Query string | `new URL(request.url).searchParams` |
| Headers | `headers()` from `next/headers` |
| Cookies | `cookies()` from `next/headers` |
| URL params | `context.params` (dynamic segments) |

# 5. Returning Responses

Use `NextResponse` for full control over status, headers, and body.

```tsx
import { NextResponse } from 'next/server'

// JSON response
return NextResponse.json({ data: result })

// JSON with custom status
return NextResponse.json({ error: 'Not found' }, { status: 404 })

// Empty response (delete success)
return new NextResponse(null, { status: 204 })

// Redirect
return NextResponse.redirect(new URL('/login', request.url))

// Custom headers
return NextResponse.json(data, {
  headers: {
    'Cache-Control': 'public, s-maxage=60',
  },
})
```

| Response type | Code |
| ------------- | ---- |
| Success with data | `NextResponse.json(data, { status: 200 })` |
| Created | `NextResponse.json(data, { status: 201 })` |
| Bad request | `NextResponse.json({ error: '...' }, { status: 400 })` |
| Unauthorized | `NextResponse.json({ error: '...' }, { status: 401 })` |
| Not found | `NextResponse.json({ error: '...' }, { status: 404 })` |
| Server error | `NextResponse.json({ error: '...' }, { status: 500 })` |

# 6. Route Handlers vs Server Actions

| Feature | Route Handler | Server Action |
| ------- | ------------- | ------------- |
| File | `route.ts` | `'use server'` function |
| Called from | `fetch`, mobile apps, webhooks | Forms, client/server components |
| HTTP methods | GET, POST, PUT, DELETE, etc. | POST only (RPC-style) |
| REST API | ✅ Natural fit | ❌ Not REST |
| Form submission | Manual fetch required | Native `<form action={...}>` |
| External clients | ✅ Any HTTP client | ❌ Next.js app only |
| Progressive enhancement | Needs client JS for fetch | Works without JS |

```tsx
// Route Handler — good for REST API consumed by mobile app
// app/api/tasks/route.ts
export async function GET() {
  const tasks = await db.task.findMany()
  return NextResponse.json(tasks)
}

// Server Action — good for form in your Next.js app
// app/actions/tasks.ts
'use server'
export async function createTask(formData: FormData) {
  const title = formData.get('title') as string
  await db.task.create({ data: { title } })
  revalidatePath('/tasks')
}
```

**Rule of thumb:** Server Actions for your app's forms; Route Handlers for APIs consumed outside your React tree.

# 7. Connecting to External APIs

Route Handlers can proxy or transform external API responses.

```tsx
// app/api/weather/route.ts
import { NextRequest, NextResponse } from 'next/server'

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url)
  const city = searchParams.get('city')

  if (!city) {
    return NextResponse.json({ error: 'City is required' }, { status: 400 })
  }

  const apiKey = process.env.WEATHER_API_KEY
  if (!apiKey) {
    return NextResponse.json({ error: 'Server misconfigured' }, { status: 500 })
  }

  try {
    const res = await fetch(
      `https://api.weatherapi.com/v1/current.json?key=${apiKey}&q=${city}`
    )

    if (!res.ok) {
      return NextResponse.json({ error: 'Weather service unavailable' }, { status: 502 })
    }

    const data = await res.json()
    return NextResponse.json({
      city: data.location.name,
      temp: data.current.temp_c,
      condition: data.current.condition.text,
    })
  } catch (error) {
    console.error('Weather fetch failed:', error)
    return NextResponse.json({ error: 'Internal server error' }, { status: 500 })
  }
}
```

| Benefit of proxying | Explanation |
| ------------------- | ----------- |
| Hide API keys | Key stays on server in `process.env` |
| Transform response | Return only fields your UI needs |
| Rate limiting | Control access before hitting external API |

# 8. Error Handling and Validation

Always validate input and return consistent error shapes.

```tsx
// lib/validations/product.ts
interface CreateProductInput {
  name: string
  price: number
  category: string
}

function validateProductInput(body: unknown): CreateProductInput | null {
  if (!body || typeof body !== 'object') return null

  const { name, price, category } = body as Record<string, unknown>

  if (typeof name !== 'string' || name.trim().length === 0) return null
  if (typeof price !== 'number' || price <= 0) return null
  if (typeof category !== 'string') return null

  return { name: name.trim(), price, category }
}
```

```tsx
// app/api/products/route.ts
export async function POST(request: NextRequest) {
  try {
    const body = await request.json()
    const input = validateProductInput(body)

    if (!input) {
      return NextResponse.json(
        { error: 'Invalid input', fields: ['name', 'price', 'category'] },
        { status: 400 }
      )
    }

    const product = await db.product.create({ data: input })
    return NextResponse.json(product, { status: 201 })
  } catch (error) {
    console.error('Create product failed:', error)
    return NextResponse.json({ error: 'Internal server error' }, { status: 500 })
  }
}
```

| Status | Meaning | Client action |
| ------ | ------- | ------------- |
| 400 | Bad input | Show validation message |
| 401 | Not authenticated | Redirect to login |
| 403 | Forbidden | Show access denied |
| 404 | Not found | Show not found UI |
| 500 | Server error | Show generic error, log details server-side |

# 9. CORS for External Clients

If mobile apps or other domains call your Route Handlers, configure CORS.

```tsx
// app/api/public/data/route.ts
import { NextRequest, NextResponse } from 'next/server'

const corsHeaders = {
  'Access-Control-Allow-Origin': 'https://partner-app.example.com',
  'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
  'Access-Control-Allow-Headers': 'Content-Type, Authorization',
}

export async function OPTIONS() {
  return new NextResponse(null, { status: 204, headers: corsHeaders })
}

export async function GET() {
  const data = await fetchPublicData()
  return NextResponse.json(data, { headers: corsHeaders })
}
```

# 10. Webhook Example

Route Handlers are ideal for receiving webhooks from Stripe, GitHub, etc.

```tsx
// app/api/webhooks/stripe/route.ts
import { NextRequest, NextResponse } from 'next/server'
import Stripe from 'stripe'

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!)

export async function POST(request: NextRequest) {
  const body = await request.text()
  const signature = request.headers.get('stripe-signature')

  if (!signature) {
    return NextResponse.json({ error: 'Missing signature' }, { status: 400 })
  }

  try {
    const event = stripe.webhooks.constructEvent(
      body,
      signature,
      process.env.STRIPE_WEBHOOK_SECRET!
    )

    if (event.type === 'checkout.session.completed') {
      const session = event.data.object
      await fulfillOrder(session.id)
    }

    return NextResponse.json({ received: true })
  } catch (error) {
    console.error('Webhook verification failed:', error)
    return NextResponse.json({ error: 'Invalid signature' }, { status: 400 })
  }
}
```

# 11. Best Practices

✔ Use Route Handlers for REST APIs, webhooks, and external client access

✔ Prefer Server Actions for form mutations inside your Next.js app

✔ Always validate and sanitize request body and query params

✔ Return consistent JSON error shapes with appropriate status codes

✔ Keep secrets in environment variables — never in response bodies

✔ Use `try/catch` and log errors server-side without leaking internals

✔ Organize handlers under `app/api/` with clear folder structure
